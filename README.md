Currently, EasyPy can only do print() and input()

# Show me an example of a code, please.
Sure! Here it is:

from easypy import prn, ipt

def example1(text):
	prn(text)

def example2(text):
	a = ipt(text)
 
	return a

example1("Hello EasyPy!")

a = example2("What is your name?")

prn(f"Your name is: {a}")
